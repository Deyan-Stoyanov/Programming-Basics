using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApp1
{
    class Program
    {
        static void Main(string[] args)
        {
            var examHour = int.Parse(Console.ReadLine());
            var examMinute = int.Parse(Console.ReadLine());
            var arrivalHour = int.Parse(Console.ReadLine());
            var arrivalMinute = int.Parse(Console.ReadLine());

            var examTotal = examHour * 60 + examMinute;
            var arrivalTotal = arrivalHour * 60 + arrivalMinute;

            if (arrivalTotal > examTotal)
            {
                Console.WriteLine("late");
                if (arrivalTotal - examTotal < 60)
                {
                    Console.WriteLine($"{arrivalTotal - examTotal} minutes after the start");
                }
                else
                {
                    if ((arrivalTotal - examTotal) % 60 <= 9) 
                    {
                        Console.WriteLine($"{(arrivalTotal - examTotal) / 60}:0{(arrivalTotal - examTotal) % 60} hours after the start");
                    }
                    else
                    {
                        Console.WriteLine($"{(arrivalTotal - examTotal) / 60}:{(arrivalTotal - examTotal) % 60} hours after the start");
                    }
                }
            }
            else if (arrivalTotal <= examTotal)
            {
                if (examTotal - arrivalTotal > 30)
                {
                    Console.WriteLine("early");
                    if ((examTotal - arrivalTotal) % 60 <= 9)
                    {
                        Console.WriteLine($"{(examTotal - arrivalTotal) / 60}:0{(examTotal - arrivalTotal) % 60} hours before the start");
                    }
                    if (examTotal - arrivalTotal < 60)
                    {
                        Console.WriteLine($"{(examTotal - arrivalTotal)} minutes before the start");

                    }
                    else
                    {
                        Console.WriteLine($"{(examTotal - arrivalTotal) / 60}:{(examTotal - arrivalTotal) % 60} hours before the start");

                    }
                }
                else
                {
                    Console.WriteLine("on time");
                    Console.WriteLine($"{(examTotal - arrivalTotal)} minutes before the start");
                }
            }

        }        
    }
}

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApp1
{
    class Program
    {
        static void Main(string[] args)
        {
            var num1 = double.Parse(Console.ReadLine());
            var num2 = double.Parse(Console.ReadLine());
            var sign = Console.ReadLine();
            var result = 0.0;

            switch (sign)
            {
                case "+":
                    result = num1 + num2;
                    if (result % 2 == 0)
                    {
                        Console.WriteLine($"{num1} + {num2} = {result} - even");
                    }
                    else
                    {
                        Console.WriteLine($"{num1} + {num2} = {result} - odd");
                    }
                    break;
                case "-":
                    result = num1 - num2;
                    if (result % 2 == 0)
                    {
                        Console.WriteLine($"{num1} - {num2} = {result} - even");
                    }
                    else
                    {
                        Console.WriteLine($"{num1} - {num2} = {result} - odd");
                    }
                    break;
                case "*":
                    result = num1 * num2;
                    if (result % 2 == 0)
                    {
                        Console.WriteLine($"{num1} * {num2} = {result} - even");
                    }
                    else
                    {
                        Console.WriteLine($"{num1} * {num2} = {result} - odd");
                    }
                    break;
                case "/":
                    if (num2 == 0)
                    {
                        Console.WriteLine($"Cannot divide {num1} by zero");
                    }
                    else
                    {
                        result = num1 / num2;
                        Console.WriteLine($"{num1} / {num2} = {result:f2}");
                    }
                    break;
                case "%":
                    if (num2 == 0)
                    {
                        Console.WriteLine($"Cannot divide {num1} by zero");
                    }
                    else
                    {
                        result = num1 % num2;
                        Console.WriteLine($"{num1} % {num2} = {result}");
                    }
                    break;
                default:
                    break;
            }
        }        
    }
}

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApp1
{
    class Program
    {
        static void Main(string[] args)
        {
            var budget = double.Parse(Console.ReadLine());
            var category = Console.ReadLine().ToLower();
            var people = int.Parse(Console.ReadLine());
            var transport = 0.0;
            

            if (people <= 4)
            {
                transport = 0.75 * budget;
            }
            else if (people <= 9)
            {
                transport = 0.6 * budget;
            }
            else if (people <= 24)
            {
                transport = 0.5 * budget;
            }
            else if (people <= 49)
            {
                transport = 0.4 * budget;
            }
            else
            {
                transport = 0.25 * budget;
            }
            var netto = budget - transport;
            switch (category)
            {
                case "normal":
                    if (netto > (people * 249.99))
                    {
                        Console.WriteLine($"Yes! You have {(netto - (people * 249.99)):f2} leva left.");
                    }
                    else
                    {
                        Console.WriteLine($"Not enough money! You need {(Math.Abs(netto - (people * 249.99))):f2} leva.");
                    }
                    break;
                case "vip":
                    if (netto > (people * 499.99))
                    {
                        Console.WriteLine($"Yes! You have {(netto - (people * 499.99)):f2} leva left.");
                    }
                    else
                    {
                        Console.WriteLine($"Not enough money! You need {(Math.Abs(netto - (people * 499.99))):f2} leva.");
                    }
                    break;
                default:
                    break;
            }
        }        
    }
}

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApp1
{
    class Program
    {
        static void Main(string[] args)
        {
            var month = Console.ReadLine().ToLower();
            var days = int.Parse(Console.ReadLine());
            var studioPrice = 0.0;
            var appartementPrice = 0.0;

            switch (month)
            {
                case "may":
                case "october":
                    studioPrice = 50;
                    appartementPrice = 65;
                    if (days > 14)
                    {
                        studioPrice = (studioPrice - (0.3 * studioPrice));
                        appartementPrice = (appartementPrice - (0.1 * appartementPrice));
                    }
                    else if (days > 7)
                    {
                        studioPrice = (studioPrice - (0.05 * studioPrice));
                    }
                    break;
                case "june":
                case "september":
                    studioPrice = 75.2;
                    appartementPrice = 68.7;
                    if (days > 14)
                    {
                        studioPrice = (studioPrice - (0.2 * studioPrice));
                        appartementPrice = (appartementPrice - (0.1 * appartementPrice));
                    }
                    break;
                case "july":
                case "august":
                    studioPrice = 76;
                    appartementPrice = 77;
                    if (days > 14)
                    {
                        appartementPrice = (appartementPrice - (0.1 * appartementPrice));
                    }
                    break;
                default:
                    break;
            }
            Console.WriteLine($"Apartment: {(appartementPrice * days):f2} lv.");
            Console.WriteLine($"Studio: {(studioPrice * days):f2} lv.");
        }        
    }
}
