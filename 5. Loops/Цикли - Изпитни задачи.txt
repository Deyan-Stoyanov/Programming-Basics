using System.Text;
using System.Threading.Tasks;

namespace ConsoleApp1
{
    class Program
    {
        static void Main(string[] args)
        {
            var n = int.Parse(Console.ReadLine());
            var leftSum = 0;
            var rightSum = 0;
            for (int i = 1; i <= 2 * n; i++)
            {
                var numbers = int.Parse(Console.ReadLine());
                if (i <= n)
                {
                    leftSum += numbers;
                }
                else
                {
                    rightSum += numbers;
                }
            }
            if (leftSum == rightSum)
            {
                Console.WriteLine($"Yes, sum = {leftSum}");
            }
            else
            {
                Console.WriteLine($"No, diff = {Math.Abs(leftSum - rightSum)}");
            }
        }
    }

            int firstPair = 0;
            int secondPair = 0;
            var maxDiff = 0;
            for (int i = 0; i < n; i++)
            {
                secondPair = firstPair;
                var firstNumber = int.Parse(Console.ReadLine());
                var secondNumber = int.Parse(Console.ReadLine());
                firstPair = firstNumber + secondNumber;

                if (i > 0)
                {
                    var currentDiff = Math.Abs(secondPair - firstPair);
                    if (currentDiff > maxDiff)
                    {
                        maxDiff = currentDiff;
                    }
                }
            }
            if (maxDiff == 0)
            {
                Console.WriteLine($"Yes, value = {firstPair}");
            }
            else
            {
                Console.WriteLine($"No, maxdiff = {maxDiff}");

            }
        }

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApp1
{
    class Program
    {
        static void Main(string[] args)
        {
            var years = int.Parse(Console.ReadLine());
            var washmashinePrice = double.Parse(Console.ReadLine());
            var toyPrice = double.Parse(Console.ReadLine());
            var toys = 0;
            var money = 0.0;
            var totalSaved = 0.0;
            for (int i = 1; i <= years; i++)
            {
                if (i % 2 != 0)
                {
                    toys++;
                }
                else
                {
                    money += ((i / 2) * 10);
                }
            }
            totalSaved = (toys * toyPrice) + (money - (years / 2));
            if (totalSaved >= washmashinePrice)
            {
                Console.WriteLine($"Yes! {(totalSaved - washmashinePrice):f2}");
            }
            else
            {
                Console.WriteLine($"No! {(washmashinePrice - totalSaved):f2}");
            }
        }
    }
