                var n = int.Parse(Console.ReadLine());
                var isTrue = false;
                for (int i = 2; i <= Math.Sqrt(n); i++)
                {
                    if (n % i == 0)
                    {
                        isTrue = true;
                        break;
                    }
                    
                    
                }
                if (n <= 1)
                {
                    isTrue = true;
                }

                if (isTrue == false)
                {
                    Console.WriteLine("Prime");
                    break;
                }
                else
                {
                    Console.WriteLine("Not prime");
                    break;

                }
            }
        }
    }

namespace ConsoleApp1
{
    class Program
    {
        static void Main(string[] args)
        {
            var n = int.Parse(Console.ReadLine());
            int numbersOnRow = 1;
            int currentNumber = 1;
            while (true)
            {
                for (int i = 0; i < numbersOnRow; i++)
                {
                    Console.Write($"{currentNumber++} ");
                    if (currentNumber == n + 1)
                    {
                        return;
                    }
                }
                numbersOnRow++;
                Console.WriteLine();
            }
        }
    }
}

    {
        static void Main(string[] args)
        {
            var n = int.Parse(Console.ReadLine());
            for (int i = 1; i <= n; i++)
            {
                for (int a = i; a <= n; a++)
                {
                    Console.Write($"{a} ");
                }
                for (int j = n - 1; j > n - i; j--)
                {
                    Console.Write($"{j} ");
                }
                Console.WriteLine();
            }
        }
    }

                for (int j = 1; j <= n; j++)

                    for (char k = 'a'; k <= 'a' + l; k++)
