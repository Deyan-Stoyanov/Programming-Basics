using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApp1
{
    class Program
    {
        static void Main(string[] args)
        {

            var n = double.Parse(Console.ReadLine());
            var lTile = double.Parse(Console.ReadLine());
            var wTile = double.Parse(Console.ReadLine());
            var lBench = double.Parse(Console.ReadLine());
            var wBench = double.Parse(Console.ReadLine());
            var area = n * n;
            var tileArea = lTile * wTile;
            var benchArea = lBench * wBench;
            var tileNm = (area - benchArea) / tileArea;
            var time = tileNm * 0.2;
            Console.WriteLine("{0} {1}", tileNm, time);

        }
    }
    
}

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApp1
{
    class Program
    {
        static void Main(string[] args)
        {

            var workdays = double.Parse(Console.ReadLine());
            var moneyperDay = double.Parse(Console.ReadLine());
            var dollarCourse = double.Parse(Console.ReadLine());
            var moneyperMonth = workdays * moneyperDay;
            var bonus = moneyperMonth * 2.5;
            var moneyperYear = (moneyperMonth * 12) + bonus;
            var taxes = moneyperYear / 4;
            var netto = moneyperYear - taxes;
            var levaNetto = netto * dollarCourse;
            var levaNettoperDay = levaNetto / 365;
            Console.WriteLine(Math.Round(levaNettoperDay, 2));
        }
    }
    
}
